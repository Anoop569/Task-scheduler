create database taskschedule;
use taskschedule;

create table signup(username varchar(50) primary key,
password varchar(50),
contact_no varchar(15),
email varchar(50),
address varchar(50));

create table task(task_id int primary key,
username varchar(50),
title varchar(250),
description varchar(450),
due_date varchar(40),
priority varchar(30),
tags varchar(80),
status varchar(30));

create table pushtime(username varchar(50), push_time varchar(30));
